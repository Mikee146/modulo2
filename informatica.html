if scelta == "q":
    print("calcolo area quadrato")
    l = float(input())
    while l <= 0:
        print("scemo")
    a = l * l
else:
    if scelta == "t":
        print("calcolo area triangolo")
        h = float(input())
        while h <= 0:
            print("insulto")
            h = float(input())
        print("calcolo l'area")
        b = float(input())
        while b <= 0:
            print("insulto")
            b = float(input())
        a = b * h / 2
    else:
        print("insulto")
        a = 0
